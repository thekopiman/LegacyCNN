343template <int channel, int width, typename T>
344void BatchNorm1d<channel, width, T>::forward_train(T (&input)[channel][width], T (&output)[channel][width])
345{
346 // Calculate mean E(X) first
347 for (int c = 0; c < channel; c++)
348 {
349 for (int i = 0; i < width; i++)
350 {
351 this->mean[c] += input[c][i];
352 }
353 this->mean[c] /= width;
354 }
355
356 // Calculate Population Variance Var(X) = 1/n sum(x - E(x))^2
357 for (int c = 0; c < channel; c++)
358 {
359 for (int i = 0; i < width; i++)
360 {
361 this->variance[c] += (input[c][i] - this->mean[c]) * (input[c][i] - this->mean[c]);
362 }
363 this->variance[c] /= width;
364 }
365
366 // Normalize and scale + shift
367 for (int c = 0; c < channel; c++)
368 {
369 for (int i = 0; i < width; i++)
370 {
371 output[c][i] = this->gamma[c] * (input[c][i] - this->mean[c]) / std::sqrt(this->variance[c] + this->eps) + this->beta[c];
372 }
373 }
374};
