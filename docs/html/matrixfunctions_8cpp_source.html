479template <size_t dim1, size_t dim2, typename T>
480void MatrixFunctions::Std(T (&input)[dim1][dim2], T (&output)[dim1])
481{
482 T eps = 1e-12;
483 T mean[dim1];
484 T variance[dim1];
485 for (int i = 0; i < dim1; i++)
486 {
487 mean[i] = 0;
488 variance[i] = 0;
489 }
490
491 // Calculate mean E(X) first
492 for (int c = 0; c < dim1; c++)
493 {
494 for (int i = 0; i < dim2; i++)
495 {
496 mean[c] += input[c][i];
497 }
498 mean[c] /= dim2;
499 }
500
501 // Calculate Population Variance Var(X)
502 for (int c = 0; c < dim1; c++)
503 {
504 for (int i = 0; i < dim2; i++)
505 {
506 variance[c] += (input[c][i] - mean[c]) * (input[c][i] - mean[c]);
507 }
508 variance[c] /= dim2;
509 output[c] = std::sqrt(Clamp(variance[c], eps));
510 }
511}
